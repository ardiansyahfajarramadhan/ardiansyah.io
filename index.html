Pesan Anda telah terkirim.`; document.getElementById("responseMessage").style.display = "block"; document.getElementById("successMessage").style.display = "block"; document.getElementById("contactForm").reset(); }, function(error) { // Jika gagal document.getElementById("responseMessage").innerText = "Gagal mengirim pesan, silakan coba lagi."; document.getElementById("responseMessage").style.display = "block"; document.getElementById("errorMessage").style.display = "block"; console.error("Error:", error); }); });
